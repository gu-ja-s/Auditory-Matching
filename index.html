import tkinter as tk import winsound import os def play_sound(sound_path): if os.path.exists(sound_path): winsound.PlaySound(sound_path, winsound.SND_FILENAME) def next_page(): print("다음 페이지로 넘어갑니다.") root = tk.Tk() root.title("소리 재생 버튼") root.geometry("400x300") # 상단 버튼 top_button = tk.Button(root, text="상단 버튼", width=20, height=5, bg="yellow", command=lambda: play_sound(r"C:\1homepage\dog1.wav")) top_button.pack(pady=(20, 10)) # 하단 버튼 프레임 bottom_frame = tk.Frame(root) bottom_frame.pack(side=tk.BOTTOM, pady=(10, 20)) # 버튼 간격을 위해 padx에 189 픽셀을 사용 button_padding = 189 bottom_button1 = tk.Button(bottom_frame, text="버튼 1", width=20, height=5, bg="yellow", command=lambda: play_sound(r"C:\1homepage\dog2.wav")) bottom_button1.pack(side=tk.LEFT, padx=button_padding) bottom_button2 = tk.Button(bottom_frame, text="버튼 2", width=20, height=5, bg="yellow", command=lambda: play_sound(r"C:\1homepage\trumpet.wav")) bottom_button2.pack(side=tk.LEFT, padx=button_padding) bottom_button3 = tk.Button(bottom_frame, text="버튼 3", width=20, height=5, bg="yellow", command=lambda: play_sound(r"C:\1homepage\water.wav")) bottom_button3.pack(side=tk.LEFT, padx=button_padding) # 화살표 버튼 arrow_button = tk.Button(root, text="다음 페이지", command=next_page) arrow_button.pack(pady=(10, 10)) root.mainloop()
